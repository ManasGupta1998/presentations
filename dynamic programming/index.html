Top-down approach

Recursion



fibo[0] = fibo[1] = 1
function:  fibonacci(n) 
{
          if fibo[n] has been calculated {
                  return fibo[n]
          else
                  return fibonacci(n-1)+fibonacci(n-2)
}


Bottom-Up approach

Iteration


function:  fibonacci(n) 
{
          fibo[0] = 1  
          fibo[1] = 1
          for ( i = 2;  i<=n;  i++) :
                    fibo[i] = fibo[i-1] + fibo[i-1]
          return fibo[n]
}
