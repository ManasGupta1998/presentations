QUEUE OPERATIONS


ENQUEUE(Q,x) {
   Q[Q.tail]=x;
   if(Q.tail==Q.length)
      Q.tail=1;
   else 
      Q.tail=Q.tail+1;
}

DEQUEUE(Q) {
   x=Q[Q.head];
   if(Q.head==Q.length)
      Q.head=1;
   else 
      Q.head=Q.head+1;
   return x;
}

Building MAX-Heap


MAX-HEAPIFY(A,i) {
    l=LEFT(i);
    r=RIGHT(i);
    if(l<=A.heap_size and A[l]>A[i])
      largest=l;
    else 
      largest=i;
    if(r<=A.heap-size and A[r]>A[largest])
      largest=r;
    if(largest!=i) {
      exchange A[i] with A[largest];
      MAX-HEAPIFY(A,largest);
    }
 }
BUILD-MAX-HEAP(A){
    A.heap-size=A.length;
    for(i=|_A.length/2_| downto 1)
       MAX-HEAPIFY(A,i);
 }
